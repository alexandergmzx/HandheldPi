# Automatically load overlays for detected cameras

# Automatically load overlays for detected DSI displays
# HHT: DSI auto-detect only; does not touch our SPI panel — safe to leave on.

# Automatically load initramfs files, if found

# Enable DRM VC4 V3D driver
# HHT: REQUIRED. The mipi-dbi SPI panel registers under KMS. The HDMI/vc4
# framebuffer enumerates as fb0; the ST7789V panel comes up as fb1 — the app's
# display.fb_device = "auto" finds it by name, so the two-fb split is expected.
dtoverlay=vc4-kms-v3d
max_framebuffers=2

# Don't have the firmware create an initial video= setting in cmdline.txt.
# Use the kernel's default instead.
disable_fw_kms_setup=1

# Run in 64-bit mode

# HHT: Camera Module 3 (IMX708), explicit overlay (verified on HHT-001).
# NO ,cam0 suffix: the Zero 2 W's single CSI connector is CAM1, the overlay
# default — ,cam0 targets a port this board doesn't have and the sensor is
# never probed. Verify with `dmesg | grep imx708` ("camera module ID 0x0301")
# and `rpicam-hello --list-cameras`.
dtoverlay=imx708

# --- HHT GamePi20 audio (managed by setup_audio.sh) ---
# GamePi20's onboard headphone/speaker circuit takes PWM audio from GPIO18
# (physical pin 12). GPIO19 is unconnected; routing both channels is the
# standard audremap option and leaves all gamepad buttons available.
dtparam=audio=on
dtoverlay=audremap,pins_18_19
# --- HHT GamePi20 audio end ---

# --- HHT GamePi20 display (managed by setup_display.sh) ---
# ST7789V 2.0" 320x240 SPI IPS panel via the mainline panel-mipi-dbi TinyDRM
# driver. This is the supported path on bookworm/trixie — fbcp-ili9341/rpi-fbcp are dead
# here because DispmanX was removed with the move to the KMS stack (see PLAN.md).
#
# The init sequence lives in firmware/st7789v_gamepi20.txt and is compiled to
# /lib/firmware/gamepi20.bin by scripts/setup_display.sh; the driver loads that
# blob because the first compatible string below is "gamepi20". That name is
# deliberately NOT "st7789v": the bare controller name collides with the staging
# fbtft module's spi:st7789v alias (wrong module loads, nothing binds). The
# panel_mipi_dbi driver itself is loaded via /etc/modules-load.d/hht-display.conf,
# also written by setup_display.sh.
#
# Pins (BCM):  DC=25  RESET=27  BACKLIGHT=24  on SPI0 CE0 (MOSI=10, SCLK=11).
# The panel is write-only (no MISO), hence dtparam=write-only.
#
# Tuning (nothing shown / garbage pixels / wrong colors / rotation): edit the
# firmware .txt (MADCTL 0x36, VCOMS 0xbb, INVON 0x21) and re-run setup_display.sh,
# or lower speed= to 32000000 — full table in docs/DEVICE_CONFIGURATION.md §3.4.
# cpha,cpol = SPI mode 3 — required by this module; mode 0 leaves it dark.
dtparam=spi=on
dtoverlay=mipi-dbi-spi,spi0-0,speed=48000000,cpha,cpol
dtparam=compatible=gamepi20\0panel-mipi-dbi-spi
dtparam=width=320,height=240,width-mm=40,height-mm=30
dtparam=reset-gpio=27,dc-gpio=25
dtparam=backlight-gpio=24
dtparam=write-only
# --- HHT GamePi20 display end ---

# --- HHT boot splash (managed by setup_splash.sh) ---
# Kill the firmware rainbow splash (HDMI only — the SPI panel is dark until its
# driver binds in the initramfs). Console/splash policy itself lives on the
# kernel cmdline, normalized by setup_splash.sh (production splash vs --diag
# verbose console) — see docs/DEVICE_CONFIGURATION.md §3.4.1.
disable_splash=1
# --- HHT boot splash end ---
