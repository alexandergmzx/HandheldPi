# --- HHT GamePi20 display (managed by setup_display.sh) ---
# ST7789V 2.0" 320x240 SPI IPS panel via the mainline panel-mipi-dbi TinyDRM
# driver. This is the supported path on bookworm/trixie — fbcp-ili9341/rpi-fbcp are dead
# here because DispmanX was removed with the move to the KMS stack (see PLAN.md).
#
# The init sequence lives in firmware/st7789v_gamepi20.txt and is compiled to
# /lib/firmware/gamepi20.bin by scripts/setup_display.sh; the driver loads that
# blob because the first compatible string below is "gamepi20". That name is
# deliberately NOT "st7789v": the bare controller name collides with the staging
# fbtft module's spi:st7789v alias (wrong module loads, nothing binds). The
# panel_mipi_dbi driver itself is loaded via /etc/modules-load.d/hht-display.conf,
# also written by setup_display.sh.
#
# Pins (BCM):  DC=25  RESET=27  BACKLIGHT=24  on SPI0 CE0 (MOSI=10, SCLK=11).
# The panel is write-only (no MISO), hence dtparam=write-only.
#
# Tuning (nothing shown / garbage pixels / wrong colors / rotation): edit the
# firmware .txt (MADCTL 0x36, VCOMS 0xbb, INVON 0x21) and re-run setup_display.sh,
# or lower speed= to 32000000 — full table in docs/DEVICE_CONFIGURATION.md §3.4.
dtparam=spi=on
dtoverlay=mipi-dbi-spi,spi0-0,speed=48000000
dtparam=compatible=gamepi20\0panel-mipi-dbi-spi
dtparam=width=320,height=240
dtparam=reset-gpio=27,dc-gpio=25
dtparam=backlight-gpio=24
dtparam=write-only
# --- HHT GamePi20 display end ---
